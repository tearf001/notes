function convert<Value extends string|number>(value: Value) : Value extends string ? number: Value extends number? string: never   {
    if(typeof value ==='string')
        return parseInt(value)
    return String(value)
}

const c = convert(1) // string
const c2 = convert('1') // number


enum SelectionKind {
  Single,
  Multiple
}

async function showQuickPickSmartExhausing<S extends SelectionKind>(
  prompt: string,
  selectionKind:S,
  items:readonly string[],
):Promise<S extends SelectionKind.Single ? string : S extends SelectionKind.Multiple ? string[]: never>{

  if (selectionKind === SelectionKind.Single)
    return "hello";
  return ["sup","nerds","pls","subscribe"]
}
// 更智能的方式
type SelectReturn  = {
	[SelectionKind.Single] : string;
	[SelectionKind.Multiple] : string[];	
}
async function showQuickPickSmart<S extends SelectionKind>(
  prompt: string,
  selectionKind:S,
  items:readonly string[],
):Promise<SelectReturn[S]>{

  if (selectionKind === SelectionKind.Single)
    return "hello";
  return ["sup","nerds","pls","subscribe"]
}
const smart = await showQuickPickSmart('Select', SelectionKind.Single, [])
const smarts = await showQuickPickSmart('Select', SelectionKind.Multiple, [])
